AI Find the length of the longest substring without repeating characters. 

def longest_unique(s):
    seen = {}
    start = result = 0
    for i, ch in enumerate(s):
        if ch in seen and seen[ch] >= start:
            start = seen[ch] + 1
        seen[ch] = i
        result = max(result, i - start + 1)
    return result
